51 template <typename, typename = void>
52 struct is_serializable : std::false_type
53 {
54 };

56 template <typename T>
57 struct is_serializable<
58 T, std::void_t<decltype(std::declval<const T>().save(
59 std::declval<std::ostream&>())),
60 decltype(std::declval<T>().load(
61 std::declval<std::istream&>()))>>
62 : std::true_type
63 {
64 };

66 template <typename T>
67 inline constexpr bool is_serializable_v = is_serializable<T>::value;

72 template <typename T>
73 std::enable_if_t<is_serializable_v<T>, std::vector<uint8_t>>
74 serialize(const T& obj)
75 {
76 std::stringstream ss;
77 obj.save(ss);
78 return compress(to_buffer(ss));
79 }
80
84 template <typename T>
85 std::enable_if_t<is_serializable_v<T>, T>
86 deserialize(const std::vector<uint8_t>& buffer)
87 {
88 std::stringstream ss;
89 from_buffer(ss, decompress(buffer));
90 T obj;
91 obj.load(ss);
92 return obj;
93 }

98 template <typename T>
99 void save_to_file(const T& obj, const std::string& filename)
100 {
101 auto data = serialize(obj);
102 uint64_t size = data.size();
103
104 std::ofstream ofs(filename, std::ios::binary);
105 if (!ofs)
106 throw std::runtime_error("Cannot open file for writing: " +
107 filename);
108 ofs.write(reinterpret_cast<const char*>(&size), sizeof(size));
109 ofs.write(reinterpret_cast<const char*>(data.data()), size);
110 }
111
115 template <typename T> T load_from_file(const std::string& filename)
116 {
117 std::ifstream ifs(filename, std::ios::binary);
118 if (!ifs)
119 throw std::runtime_error("Cannot open file for reading: " +
120 filename);
121
122 uint64_t size;
123 ifs.read(reinterpret_cast<char*>(&size), sizeof(size));
124 std::vector<uint8_t> buffer(size);
125 ifs.read(reinterpret_cast<char*>(buffer.data()), size);
126
127 return deserialize<T>(buffer);
128 }
